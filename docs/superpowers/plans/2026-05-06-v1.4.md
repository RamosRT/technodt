```python
"""Background 1C timestamp marking — fires as asyncio tasks, never blocks API response."""
import asyncio
import logging
import uuid
from datetime import datetime

from app.models.envelope_document import EnvelopeDocument
from app.services.audit import write_event
from app.services.odata import (
    MARK_ELIGIBLE_ENTITIES,
    PROP_REGISTERED,
    PROP_SEALED,
    PROP_VERIFIED,
    OneCClient,
)

log = logging.getLogger(__name__)


async def _mark_documents_with_retry(
    one_c: OneCClient,
    marks: list[tuple[uuid.UUID, str, uuid.UUID, datetime]],
    session_factory,
) -> None:
    async def _attempt() -> None:
        await asyncio.gather(*[
            one_c.mark_document(doc_guid, entity, prop_key, value)
            for doc_guid, entity, prop_key, value in marks
        ])

    try:
        await _attempt()
        return
    except Exception as exc:
        log.warning("1C marks first attempt failed: %s — retrying in 5s", exc)

    await asyncio.sleep(5)
    try:
        await _attempt()
    except Exception as exc:
        log.warning("1C marks retry failed: %s", exc)
        async with session_factory() as session:
            for doc_guid, _, prop_key, _ in marks:
                await write_event(
                    session,
                    envelope_id=None,
                    event="onec_mark_failed",
                    actor="system",
                    payload={
                        "doc_guid": str(doc_guid),
                        "property_key": str(prop_key),
                        "error": str(exc),
                    },
                )
            await session.commit()


def fire_seal_marks(
    one_c: OneCClient,
    docs: list[EnvelopeDocument],
    sealed_at: datetime,
    session_factory,
    *,
    enabled: bool = True,
) -> None:
    if not enabled:
        return
    eligible = [d for d in docs if d.doc_entity in MARK_ELIGIBLE_ENTITIES]
    if not eligible:
        return
    marks: list[tuple[uuid.UUID, str, uuid.UUID, datetime]] = []
    for doc in eligible:
        marks.append((doc.doc_guid, doc.doc_entity, PROP_REGISTERED, doc.added_at))
        marks.append((doc.doc_guid, doc.doc_entity, PROP_SEALED, sealed_at))
    asyncio.create_task(_mark_documents_with_retry(one_c, marks, session_factory))


def fire_verify_marks(
    one_c: OneCClient,
    docs: list[EnvelopeDocument],
    verified_at: datetime,
    session_factory,
    *,
    enabled: bool = True,
) -> None:
    if not enabled:
        return
    eligible = [
        d for d in docs
        if d.doc_entity in MARK_ELIGIBLE_ENTITIES and d.scanned_at_verification is not None
    ]
    if not eligible:
        return
    marks = [(d.doc_guid, d.doc_entity, PROP_VERIFIED, verified_at) for d in eligible]
    asyncio.create_task(_mark_documents_with_retry(one_c, marks, session_factory))
```
