# ТехноКонверт v1.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add operator registry with roles, persistent navbar, envelope/document list views, unseal operation, audit log UI, and DB reset button on top of the existing MVP.

**Architecture:** All features share the single FastAPI + SQLAlchemy async backend. Auth is upgraded to a DB-backed operator table; the existing cookie/token approach is replaced by DB lookup. UI remains HTMX partials served by Jinja2; the navbar is added to `base.html` and each new view is a partial swapped into `#main-area`. No new services are added for HTTP clients — only new SQLAlchemy queries.

**Tech Stack:** Python 3.14, FastAPI, SQLAlchemy async, Alembic, PostgreSQL 16, HTMX, Jinja2, Lucide Icons CDN, pytest + pytest-asyncio + httpx.AsyncClient.

```python
import uuid

from sqlalchemy import select
from sqlalchemy.exc import IntegrityError
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.operator import Operator


async def ensure_operator(
    session: AsyncSession,
    username: str,
    *,
    bootstrap: bool = False,
) -> Operator:
    op = (
        await session.execute(select(Operator).where(Operator.username == username))
    ).scalar_one_or_none()

    if op is None:
        op = Operator(username=username, is_admin=bootstrap)
        session.add(op)
        try:
            await session.flush()
        except IntegrityError:
            await session.rollback()
            op = (
                await session.execute(select(Operator).where(Operator.username == username))
            ).scalar_one()
            if bootstrap and not op.is_admin:
                op.is_admin = True
                await session.flush()
    elif bootstrap and not op.is_admin:
        op.is_admin = True
        await session.flush()

    return op


async def list_operators(session: AsyncSession) -> list[Operator]:
    result = await session.execute(select(Operator).order_by(Operator.username))
    return list(result.scalars().all())


async def patch_operator(
    session: AsyncSession,
    *,
    operator_id: uuid.UUID,
    is_admin: bool | None,
    is_active: bool | None,
) -> Operator:
    op = (
        await session.execute(select(Operator).where(Operator.id == operator_id))
    ).scalar_one()
    if is_admin is not None:
        op.is_admin = is_admin
    if is_active is not None:
        op.is_active = is_active
    await session.flush()
    return op
```

```python
from urllib.parse import unquote

from fastapi import Cookie, Depends
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.config import get_settings
from app.db import get_session
from app.exceptions import AdminTokenInvalid, OperatorRequired
from app.models.operator import Operator
from app.services.operators import ensure_operator


def require_operator():
    async def dep(
        operator_name: str | None = Cookie(default=None),
        session: AsyncSession = Depends(get_session),
    ) -> str:
        if not operator_name:
            raise OperatorRequired("Введите имя оператора")
        name = unquote(operator_name)
        settings = get_settings()
        is_bootstrap = bool(settings.bootstrap_admin) and name == settings.bootstrap_admin
        op = await ensure_operator(session, name, bootstrap=is_bootstrap)
        if not op.is_active:
            raise OperatorRequired("Оператор деактивирован")
        return name

    return Depends(dep)


def require_admin():
    async def dep(
        operator_name: str | None = Cookie(default=None),
        session: AsyncSession = Depends(get_session),
    ) -> None:
        if not operator_name:
            raise AdminTokenInvalid("Требуется авторизация администратора")
        name = unquote(operator_name)
        op = (
            await session.execute(
                select(Operator).where(
                    Operator.username == name,
                    Operator.is_admin.is_(True),
                    Operator.is_active.is_(True),
                )
            )
        ).scalar_one_or_none()
        if op is None:
            raise AdminTokenInvalid("Нет прав администратора")

    return Depends(dep)


async def get_is_admin(
    operator_name: str | None = Cookie(default=None),
    session: AsyncSession = Depends(get_session),
) -> bool:
    if not operator_name:
        return False
    name = unquote(operator_name)
    op = (
        await session.execute(
            select(Operator).where(
                Operator.username == name,
                Operator.is_admin.is_(True),
                Operator.is_active.is_(True),
            )
        )
    ).scalar_one_or_none()
    return op is not None
```

from datetime import date as DateType
from app.models import Branch


async def list_envelopes(
    session: AsyncSession,
    *,
    status: EnvelopeStatus | None = None,
    date_from: DateType | None = None,
    date_to: DateType | None = None,
    origin_branch_id: uuid.UUID | None = None,
    search: str | None = None,
    page: int = 1,
    page_size: int = 25,
) -> tuple[list[Envelope], int]:
    from sqlalchemy import func as sqlfunc, cast, Date

    stmt = select(Envelope)
    if status is not None:
        stmt = stmt.where(Envelope.status == status)
    if date_from is not None:
        stmt = stmt.where(cast(Envelope.created_at, Date) >= date_from)
    if date_to is not None:
        stmt = stmt.where(cast(Envelope.created_at, Date) <= date_to)
    if origin_branch_id is not None:
        stmt = stmt.where(Envelope.origin_branch_id == origin_branch_id)
    if search:
        stmt = stmt.where(
            Envelope.number.ilike(f"%{search}%") | Envelope.barcode.ilike(f"%{search}%")
        )

    count_stmt = select(sqlfunc.count()).select_from(stmt.subquery())
    total: int = (await session.execute(count_stmt)).scalar_one()

    stmt = stmt.order_by(Envelope.created_at.desc())
    stmt = stmt.offset((page - 1) * page_size).limit(page_size)
    stmt = stmt.options(selectinload(Envelope.documents))

    rows = list((await session.execute(stmt)).scalars().all())

from sqlalchemy import select, func, and_, cast, Date

from app.models import Envelope, EnvelopeDocument, EnvelopeStatus


class DocumentRow:
    def __init__(
        self,
        doc: EnvelopeDocument,
        envelope: Envelope,
    ):
        self.doc = doc
        self.envelope = envelope

    @property
    def checkpoint_status(self) -> str:
        if self.doc.scanned_at_verification is not None:
            return "verified"
        if self.envelope.status in (EnvelopeStatus.verified, EnvelopeStatus.verified_with_discrepancy):
            return "discrepancy"
        if self.envelope.status == EnvelopeStatus.sealed:
            return "in_transit"
        return "draft"


async def list_documents(
    session: AsyncSession,
    *,
    date_from: DateType | None = None,
    date_to: DateType | None = None,
    doc_type: str | None = None,
    status: str | None = None,
    origin_branch_id: uuid.UUID | None = None,
    search: str | None = None,
    page: int = 1,
    page_size: int = 50,
) -> tuple[list[DocumentRow], int, dict]:
    stmt = (
        select(EnvelopeDocument, Envelope)
        .join(Envelope, EnvelopeDocument.envelope_id == Envelope.id)
    )

    if date_from is not None:
        stmt = stmt.where(cast(EnvelopeDocument.added_at, Date) >= date_from)
    if date_to is not None:
        stmt = stmt.where(cast(EnvelopeDocument.added_at, Date) <= date_to)
    if doc_type is not None:
        stmt = stmt.where(EnvelopeDocument.doc_kind == doc_type)
    if origin_branch_id is not None:
        stmt = stmt.where(Envelope.origin_branch_id == origin_branch_id)
    if search:
        stmt = stmt.where(
            EnvelopeDocument.doc_number.ilike(f"%{search}%")
            | EnvelopeDocument.doc_barcode.ilike(f"%{search}%")
        )

    if status == "verified":
        stmt = stmt.where(EnvelopeDocument.scanned_at_verification.is_not(None))
    elif status == "discrepancy":
        stmt = stmt.where(
            EnvelopeDocument.scanned_at_verification.is_(None),
            Envelope.status == EnvelopeStatus.verified_with_discrepancy,
        )
    elif status == "in_transit":
        stmt = stmt.where(
            EnvelopeDocument.scanned_at_verification.is_(None),
            Envelope.status == EnvelopeStatus.sealed,
        )

    count_stmt = select(func.count()).select_from(stmt.subquery())
    total: int = (await session.execute(count_stmt)).scalar_one()

    stmt = stmt.order_by(EnvelopeDocument.added_at.desc())
    stmt = stmt.offset((page - 1) * page_size).limit(page_size)

    rows_raw = (await session.execute(stmt)).all()
    rows = [DocumentRow(doc=r[0], envelope=r[1]) for r in rows_raw]

    # Summary counts (always over the full filter, not just this page)
    summary_stmt = (
        select(
            func.count().label("total"),
            func.count(EnvelopeDocument.scanned_at_verification).label("verified"),
        )
        .select_from(
            select(EnvelopeDocument, Envelope)
            .join(Envelope, EnvelopeDocument.envelope_id == Envelope.id)
            .subquery()
        )
    )
    summary_row = (await session.execute(summary_stmt)).one()
    summary = {
        "total": total,
        "verified": summary_row.verified,
        "in_transit": 0,
        "discrepancy": 0,
    }

    return rows, total, summary


def build_csv(rows: list[DocumentRow]) -> str:
    buf = io.StringIO()
    writer = csv.writer(buf)
    writer.writerow([
        "Тип", "№ документа", "Дата документа",
        "Конверт", "Добавлен", "Запечатан", "Верифицирован", "Статус",
    ])
    for r in rows:
        writer.writerow([
            r.doc.doc_kind or "",
            r.doc.doc_number or "",
            r.doc.doc_date.isoformat() if r.doc.doc_date else "",
            r.envelope.number,
            r.doc.added_at.strftime("%d.%m.%Y %H:%M") if r.doc.added_at else "",
            r.envelope.sealed_at.strftime("%d.%m.%Y %H:%M") if r.envelope.sealed_at else "",
            r.doc.scanned_at_verification.strftime("%d.%m.%Y %H:%M") if r.doc.scanned_at_verification else "",
            r.checkpoint_status,
        ])
    return buf.getvalue()

```python
@router.get("/ui/envelopes-list/results", response_class=HTMLResponse)
async def ui_envelopes_list_results(
    request: Request,
    status: str | None = None,
    date_from: str | None = None,
    date_to: str | None = None,
    origin_branch_id: uuid.UUID | None = None,
    search: str | None = None,
    page: int = 1,
    page_size: int = 25,
    session: AsyncSession = Depends(get_session),
    operator: str | None = Depends(_operator),
    is_admin: bool = Depends(get_is_admin),
):
    from datetime import date as DateType
    from app.models import EnvelopeStatus as ES
    from app.services.envelopes import list_envelopes

    status_enum = ES(status) if status else None
    df = DateType.fromisoformat(date_from) if date_from else None
    dt = DateType.fromisoformat(date_to) if date_to else None

    envelopes, total = await list_envelopes(
        session, status=status_enum, date_from=df, date_to=dt,
        origin_branch_id=origin_branch_id, search=search,
        page=page, page_size=page_size,
    )
    branches = await dict_svc.list_branches(session, only_active=True)
    return templates.TemplateResponse(request, "partials/envelopes_list.html", {
        "operator": operator, "is_admin": is_admin,
        "envelopes": envelopes, "total": total,
        "page": page, "page_size": page_size, "branches": branches,
    })
```
